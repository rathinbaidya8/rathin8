How to add chatbot to your HTML file — step by step guide

1
CSS — <head> tag এর ভেতরে বসাও
তোমার file এ </head> tag খোঁজো, তার ঠিক আগে এই CSS টা paste করো:
<style>
  /* Chatbot button */
  .chat-bubble { position:fixed; bottom:24px; right:24px;
    width:54px; height:54px; border-radius:50%;
    background:#7F77DD; color:#fff; border:none;
    font-size:22px; cursor:pointer; z-index:999; }

  /* Chatbot window */
  .chat-window { position:fixed; bottom:90px; right:24px;
    width:300px; background:#fff; border-radius:12px;
    border:1px solid #ddd; display:none; z-index:999;
    flex-direction:column; overflow:hidden; }
  .chat-window.open { display:flex; }
  .chat-header { background:#7F77DD; color:#fff;
    padding:12px 16px; font-size:14px; font-weight:500; }
  .chat-msgs { height:220px; overflow-y:auto;
    padding:12px; display:flex; flex-direction:column; gap:8px; }
  .bot-msg { background:#f0f0f0; border-radius:10px;
    padding:8px 12px; font-size:13px; max-width:85%; }
  .user-msg { background:#7F77DD; color:#fff; border-radius:10px;
    padding:8px 12px; font-size:13px;
    max-width:85%; align-self:flex-end; }
  .chat-opts { padding:10px; border-top:1px solid #eee;
    display:flex; flex-wrap:wrap; gap:6px; }
  .opt { padding:5px 12px; border-radius:20px;
    border:1px solid #AFA9EC; background:#EEEDFE;
    color:#534AB7; font-size:12px; cursor:pointer; }
</style>
2
HTML — </body> tag এর ঠিক আগে বসাও
তোমার file এ </body> খোঁজো, তার আগে এই chatbot HTML টা paste করো:
<!-- Chatbot button -->
<button class="chat-bubble" onclick="toggleChat()">💬</button>

<!-- Chatbot window -->
<div class="chat-window" id="chatWindow">
  <div class="chat-header">💬 Help Assistant</div>
  <div class="chat-msgs" id="chatMsgs"></div>
  <div class="chat-opts" id="chatOpts"></div>
</div>
3
JavaScript — HTML এর নিচেই রাখো
Step 2 এর code এর ঠিক নিচে এই script টা বসাও। এখানেই তোমার questions & answers লেখা আছে:
<script>
var data = {
  "start": { msg: "Ki help lagbe?", opts: [
    { label: "Pricing", next: "pricing" },
    { label: "Contact", next: "contact" }]},

  "pricing": { msg: "Basic: ৳500/mo!", opts: [
    { label: "Kinbo", link: "https://..." },
    { label: "Back", next: "start" }]},

  "contact": { msg: "Email: hello@site.com", opts: [
    { label: "Back", next: "start" }]}
};

function toggleChat() {
  var w = document.getElementById('chatWindow');
  w.classList.toggle('open');
  if(w.classList.contains('open')) goTo('start');
}

function goTo(key, userText) {
  var m=document.getElementById('chatMsgs');
  var o=document.getElementById('chatOpts');
  if(userText){ var u=document.createElement('div');
    u.className='user-msg'; u.textContent=userText; m.appendChild(u); }
  var s=data[key]; if(!s) return;
  var b=document.createElement('div');
  b.className='bot-msg'; b.textContent=s.msg; m.appendChild(b);
  o.innerHTML='';
  s.opts.forEach(function(opt){
    var btn=document.createElement('button');
    btn.className='opt'; btn.textContent=opt.label;
    if(opt.link) btn.onclick=function(){window.open(opt.link)};
    else btn.onclick=function(){goTo(opt.next,opt.label)};
    o.appendChild(btn);
  });
  m.scrollTop=m.scrollHeight;
}
</script>
4
নিজের questions যোগ করো এইভাবে
data object এ নতুন block যোগ করো, তারপর কোনো option এর next এ সেই নাম দাও:
"delivery": { msg: "3-5 din e deliver hoy!", opts: [
  { label: "Order korbo", link: "https://shop.com" },
  { label: "Back", next: "start" }
]}
তোমার HTML file VS Code বা Notepad এ open করো → Ctrl+F দিয়ে </head> আর </body> খোঁজো → উপরের steps অনুযায়ী paste করো → file save করে browser এ open করো। নিচে ডানে 💬 button দেখতে পাবে!
